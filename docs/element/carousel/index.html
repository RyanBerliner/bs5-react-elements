Carousel

Wrapper for the Bootstrap carousel component.

Example Usage

import React from "react";
import {Carousel} from "bs5-react-elements";

function ExampleUsage() {
  return <Carousel
    id="example-usage"
    className="carousel slide"
    data-bs-ride="carousel"
  >
    <div className="carousel-inner">
      <div className="carousel-item active">
        Lorem ipsum 1
      </div>
      <div className="carousel-item">
        Lorem ipsum 2
      </div>
      <div className="carousel-item">
        Lorem ipsum 3
      </div>
    </div>
  </Carousel>;
}
